/* asciiToFloat: this will take ASCII input and convert it to a double */
double asciiToFloat(char s[])
{
    double val;
    double power;
    int i;
    int sign;

    // 1. Skip leading whitespace
    for (i = 0; isspace(s[i]); i++) /* gets rid of any whitespace */
        ;

    // 2. Determine the sign
    sign = (s[i] == '-') ? -1 : 1; /* sets the sign of the input */
    if (s[i] == '+' || s[i] == '-') /* excludes operands, plus and minus */
        i++;

    // 3. Convert the integer part
    for (val = 0.0; isdigit(s[i]); i++) 
         val = 10.0 * val + (s[i] - '0'); 

    // 4. Convert the fractional part
    if (s[i] == '.')  // Corrected the assignment operator to equality operator
        i++;
    for (power = 1.0; isdigit(s[i]); i++) {
        val = 10.0 * val + (s[i] - '0');
        power *= 10.0;
    }

    // 5. Combine the sign, integer part, and fractional part
    return sign * val / power;
}
